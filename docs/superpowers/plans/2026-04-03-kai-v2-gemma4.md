"""Ollama HTTP API client for kai."""
import json
from urllib.request import urlopen, Request
from urllib.error import URLError

OLLAMA_BASE = "http://localhost:11434"
MODEL = "gemma4:e4b"


def is_available() -> bool:
    """Check if Ollama is running and the model is pulled."""
    try:
        with urlopen(f"{OLLAMA_BASE}/api/tags") as resp:
            data = json.loads(resp.read())
            names = [m["name"] for m in data.get("models", [])]
            return MODEL in names
    except (URLError, Exception):
        return False

def chat(messages: list, tools: list | None = None,
         stream: bool = True):
    """Send a chat request to Ollama.

    When stream=True, yields parsed JSON chunks (NDJSON lines).
    When stream=False, returns the full parsed response dict.
    Raises ConnectionError if Ollama is unreachable.
    """
    body = {
        "model": MODEL,
        "messages": messages,
        "stream": stream,
    }
    if tools:
        body["tools"] = tools

    data = json.dumps(body).encode()
    req = Request(
        f"{OLLAMA_BASE}/api/chat",
        data=data,
        headers={"Content-Type": "application/json"},
        method="POST",
    )

    try:
        resp = urlopen(req)
    except URLError as e:
        raise ConnectionError(
            f"Cannot reach Ollama at {OLLAMA_BASE}. "
            f"Is it running? Try: ollama serve\n({e})"
        )

    if stream:
        return _stream_response(resp)
    else:
        with resp:
            return json.loads(resp.read())


def _stream_response(resp):
    """Yield parsed JSON objects from an NDJSON streaming response."""
    with resp:
        for line in resp:
            line = line.strip()
            if line:
                yield json.loads(line)
```

# tools.py
"""Tool registry and implementations for kai."""
import csv
import json
import os
import re
import subprocess
import xml.etree.ElementTree as ET
import zipfile
from pathlib import Path

# ── Constants ──

PROJECTS_DIR = Path.home() / "Projects"
MAX_READ_CHARS = 8000


# ── Text extraction (carried from v1, same logic) ──

def _extract_text(file_path: Path) -> str:
    """Extract text from various file formats using only stdlib."""
    suffix = file_path.suffix.lower()

    # Plain text / source code
    text_suffixes = (
        ".md", ".txt", ".markdown",
        ".sh", ".bash", ".zsh",
        ".py", ".ts", ".tsx", ".js", ".jsx",
        ".c", ".h", ".cpp", ".hpp",
        ".json", ".yaml", ".yml", ".toml",
        ".html", ".css", ".scss",
        ".rs", ".go", ".java", ".kt", ".swift",
        ".rb", ".pl", ".lua", ".r",
        ".sql", ".graphql",
        ".env", ".ini", ".cfg", ".conf",
        ".xml", ".plist",
    )
    if suffix in text_suffixes:
        return file_path.read_text(encoding="utf-8", errors="replace")

    # CSV
    if suffix == ".csv":
        rows = []
        with open(file_path, "r", encoding="utf-8", errors="replace") as f:
            for row in csv.reader(f):
                rows.append("\t".join(row))
        return "\n".join(rows)

    # TSV
    if suffix == ".tsv":
        rows = []
        with open(file_path, "r", encoding="utf-8", errors="replace") as f:
            for row in csv.reader(f, delimiter="\t"):
                rows.append("\t".join(row))
        return "\n".join(rows)

    # DOCX
    if suffix == ".docx":
        text_parts = []
        with zipfile.ZipFile(file_path, "r") as zf:
            if "word/document.xml" not in zf.namelist():
                return "Error: invalid DOCX (missing word/document.xml)"
            xml_content = zf.read("word/document.xml").decode("utf-8")
            root = ET.fromstring(xml_content)
            for para in root.iter():
                tag = para.tag.split("}")[-1] if "}" in para.tag else para.tag
                if tag == "p":
                    para_text = []
                    for elem in para.iter():
                        elem_tag = elem.tag.split("}")[-1] if "}" in elem.tag else elem.tag
                        if elem_tag == "t" and elem.text:
                            para_text.append(elem.text)
                        elif elem_tag == "tab":
                            para_text.append("\t")
                        elif elem_tag in ("br", "cr"):
                            para_text.append("\n")
                    if para_text:
                        text_parts.append("".join(para_text))
            return "\n".join(text_parts).strip()

    # PDF (basic)
    if suffix == ".pdf":
        content = file_path.read_bytes()
        parts = []
        for m in re.finditer(rb"\((.*?)\)", content):
            try:
                text = m.group(1).decode("latin-1", errors="ignore")
                text = "".join(
                    c if (32 <= ord(c) < 127) or c in "\n\r\t" else " "
                    for c in text
                )
                if len(text.strip()) > 2:
                    parts.append(text.strip())
            except Exception:
                pass
        if parts:
            return re.sub(r"\s+", " ", " ".join(parts)).strip()
        return "Error: could not extract text from PDF"

    # .pages (macOS)
    if suffix == ".pages":
        try:
            result = subprocess.run(
                ["textutil", "-convert", "txt", "-stdout", str(file_path)],
                capture_output=True, text=True, timeout=10,
            )
            if result.returncode == 0 and result.stdout:
                return result.stdout
        except (subprocess.TimeoutExpired, FileNotFoundError):
            pass
        return "Error: could not extract text from .pages file"

    # Fallback: try as text
    try:
        return file_path.read_text(encoding="utf-8", errors="replace")
    except Exception:
        return f"Error: unsupported file format: {suffix}"


# ── Tool implementations ──

def read_file(path: str) -> str:
    """Read a file and return its text content (capped at 8000 chars)."""
    try:
        # Fix hallucinated Linux home paths
        home = str(Path.home())
        for fake in ["/home/user/", "/home/user"]:
            if path.startswith(fake):
                path = home + "/" + path[len(fake):]
                break

        fp = Path(path).expanduser().resolve()
        if not fp.is_file():
            return f"Error: file not found: {fp}"

        content = _extract_text(fp)
        if len(content) > MAX_READ_CHARS:
            return content[:MAX_READ_CHARS] + "\n...(truncated at 8000 chars)..."
        return content
    except Exception as e:
        return f"Error reading file: {e}"
```

def write_file(path: str, content: str, yolo: bool = False) -> str:
    """Write content to a file. Asks for confirmation based on permission rules."""
    try:
        fp = Path(path).expanduser().resolve()
        exists = fp.is_file()

        # Permission check
        needs_confirm = False
        if exists:
            needs_confirm = True
            prompt_msg = f"Overwrite {fp}? [y/N] "
        elif not str(fp).startswith(str(PROJECTS_DIR)):
            needs_confirm = True
            prompt_msg = f"Write to {fp} (outside ~/Projects)? [y/N] "

        if needs_confirm and not yolo:
            answer = input(prompt_msg).strip().lower()
            if answer != "y":
                return "Cancelled."

        fp.parent.mkdir(parents=True, exist_ok=True)
        fp.write_text(content, encoding="utf-8")
        return f"Wrote {len(content)} chars to {fp}"
    except Exception as e:
        return f"Error writing file: {e}"

def _human_size(size: int) -> str:
    """Convert bytes to human-readable size."""
    for unit in ("B", "KB", "MB", "GB"):
        if size < 1024:
            return f"{size:.0f}{unit}" if unit == "B" else f"{size:.1f}{unit}"
        size /= 1024
    return f"{size:.1f}TB"


def list_files(directory: str) -> str:
    """List contents of a directory with type indicators and sizes."""
    try:
        dp = Path(directory).expanduser().resolve()
        if not dp.is_dir():
            return f"Error: not a directory: {dp}"

        entries = sorted(dp.iterdir(), key=lambda p: (not p.is_dir(), p.name.lower()))
        lines = []
        for entry in entries[:50]:
            if entry.is_dir():
                lines.append(f"  [dir]  {entry.name}/")
            else:
                size = _human_size(entry.stat().st_size)
                lines.append(f"  [file] {entry.name}  ({size})")

        result = f"{dp}/\n" + "\n".join(lines)
        if len(entries) > 50:
            result += f"\n  ... truncated ({len(entries)} total entries)"
        return result
    except Exception as e:
        return f"Error listing directory: {e}"
```

# ── Utilities ──

def timestamp() -> str:
    return datetime.now().strftime("%Y%m%d_%H%M%S")


def cap(s: str, limit: int = 4000) -> str:
    return s[:limit] + "\n...(truncated)..." if len(s) > limit else s


def slugify_words(s: str, max_words: int = 5) -> str:
    words = re.findall(r"[A-Za-z0-9]+", s.lower())
    return "-".join(words[:max_words]) if words else "memory"


# ── Session management ──

def latest_session_path() -> Path:
    SESSIONS_DIR.mkdir(parents=True, exist_ok=True)
    sessions = sorted(SESSIONS_DIR.glob("session_*.md"))
    if sessions:
        return sessions[-1]
    p = SESSIONS_DIR / f"session_{timestamp()}.md"
    p.write_text("# Session\n\n", encoding="utf-8")
    return p


def append_turn(session_file: Path, role: str, text: str) -> None:
    with session_file.open("a", encoding="utf-8") as f:
        f.write(f"\n\n## {role}\n{text.strip()}\n")


# ── Memory search ──

def search_memories(query: str, k: int = 3) -> list:
    """Keyword scoring over memories/ directory. Returns top k (path, text) pairs."""
    MEMORIES_DIR.mkdir(parents=True, exist_ok=True)
    terms = [t for t in re.findall(r"[A-Za-z0-9]+", query.lower()) if len(t) >= 3]
    if not terms:
        return []
    scored = []
    for p in MEMORIES_DIR.glob("memory_*.md"):
        try:
            txt = p.read_text(encoding="utf-8", errors="replace")
        except Exception:
            continue
        s = sum(txt.lower().count(t) for t in terms)
        if s > 0:
            scored.append((s, p, txt))
    scored.sort(key=lambda x: x[0], reverse=True)
    return [(p, txt) for _, p, txt in scored[:k]]


# ── Memory writing ──

def make_memory(session_text: str) -> tuple:
    """Ask the model to distill a session into a memory note. Returns (title, markdown)."""
    meta_prompt = (
        "You will write a memory note from the conversation below.\n\n"
        "Return in this exact format:\n"
        "TITLE: <exactly 5 words>\n"
        "SUMMARY:\n"
        "- <bullet>\n"
        "- <bullet>\n"
        "FACTS:\n"
        "- <fact>\n\n"
        "CONVERSATION:\n"
        f"{session_text}\n"
    )
    messages = [{"role": "user", "content": meta_prompt}]
    resp = ollama_client.chat(messages, stream=False)
    raw = resp["message"]["content"]
    m = re.search(r"TITLE:\s*(.+)", raw)
    title = m.group(1).strip() if m else "memory note"
    title5 = " ".join(re.findall(r"[A-Za-z0-9]+", title)[:5]) or "memory note"
    md = f"# {title5}\n\n{raw.strip()}\n"
    return title5, md


def write_memory_file(title5: str, md: str) -> Path:
    MEMORIES_DIR.mkdir(parents=True, exist_ok=True)
    fname = f"memory_{slugify_words(title5, 5)}_{timestamp()}.md"
    path = MEMORIES_DIR / fname
    path.write_text(md, encoding="utf-8")
    return path

def log_query(prompt: str, output: str, mode: str = "standard",
              tool_calls: list | None = None,
              files: list | None = None) -> Path:
    LOGS_DIR.mkdir(parents=True, exist_ok=True)
    entry = {
        "timestamp": datetime.now().isoformat(),
        "prompt": prompt,
        "output": cap(output, 4000),
        "mode": mode,
        "tool_calls": tool_calls or [],
        "files": files or [],
        "rating": None,
    }
    log_file = LOGS_DIR / f"log_{timestamp()}.json"
    log_file.write_text(json.dumps(entry, indent=2, ensure_ascii=False), encoding="utf-8")
    return log_file


def list_logs(n: int = 10) -> list:
    LOGS_DIR.mkdir(parents=True, exist_ok=True)
    log_files = sorted(LOGS_DIR.glob("log_*.json"), reverse=True)
    entries = []
    for lf in log_files[:n]:
        try:
            entry = json.loads(lf.read_text(encoding="utf-8"))
            entry["_file"] = lf.name
            entries.append(entry)
        except Exception:
            continue
    return entries


def rate_last_log(rating: int) -> Optional[Path]:

    log_files = sorted(LOGS_DIR.glob("log_*.json"), reverse=True)
    if not log_files:

    lf = log_files[0]
    try:
        entry = json.loads(lf.read_text(encoding="utf-8"))
        entry["rating"] = max(1, min(5, rating))
        lf.write_text(json.dumps(entry, indent=2, ensure_ascii=False), encoding="utf-8")
        return lf
    except Exception:
        return None

# ── Streaming ──

def stream_response(messages: list) -> str:
    """Stream a chat response to stdout, return full text."""
    full_text = ""
    for chunk in ollama_client.chat(messages, stream=True):
        content = chunk.get("message", {}).get("content", "")
        if content:
            sys.stdout.write(content)
            sys.stdout.flush()
            full_text += content
    print()  # newline after streaming
    return full_text


# ── Agentic tool loop ──

def run_with_tools(messages: list, yolo: bool = False,
                   debug: bool = False) -> tuple:
    """Run the agentic tool-calling loop.

    Returns (final_answer: str, tool_call_log: list[dict]).
    """
    tool_schemas = tool_module.get_tool_schemas()
    tool_call_log = []

    for round_num in range(MAX_ROUNDS):
        if debug:
            print(f"[{PROG_NAME}] Round {round_num + 1}/{MAX_ROUNDS}...",
                  file=sys.stderr)

        resp = ollama_client.chat(messages, tools=tool_schemas, stream=False)
        msg = resp["message"]

        # Check for tool calls
        tool_calls = msg.get("tool_calls", [])
        if not tool_calls:
            # No tool calls — this is the final answer
            return msg.get("content", ""), tool_call_log

        # Append the assistant message (with tool calls) to history
        messages.append(msg)

        # Execute each tool call
        for tc in tool_calls:
            func = tc["function"]
            tool_name = func["name"]
            tool_args = func.get("arguments", {})

            print(f"[{PROG_NAME}] Calling {tool_name}({json.dumps(tool_args)})",
                  file=sys.stderr)

            result = tool_module.execute_tool(tool_name, tool_args, yolo=yolo)

            if debug:
                print(f"[{PROG_NAME}] {tool_name} returned {len(result)} chars",
                      file=sys.stderr)

            tool_call_log.append({
                "round": round_num + 1,
                "tool": tool_name,
                "args": tool_args,
                "result_length": len(result),
            })

            # Append tool result to message history
            messages.append({
                "role": "tool",
                "content": result,
            })

    # Exhausted all rounds — return whatever content was in the last response
    return msg.get("content", ""), tool_call_log

              file=sys.stderr)
        sys.exit(1)

    # ── Heuristic triggers ──
    lower = user_text.lower()
    remember_trigger = args.remember or any(
        k in lower for k in ["remember this", "save memory", "/remember"]
    )
    use_memory_trigger = args.use_memory or any(
        k in lower for k in ["use memory", "/memory"]
    )

    # ── Session management ──
    if args.new_session:
        session_file = SESSIONS_DIR / f"session_{timestamp()}.md"
        session_file.parent.mkdir(parents=True, exist_ok=True)
        session_file.write_text("# Session\n\n", encoding="utf-8")
    else:
        session_file = latest_session_path()

    # ── File ingestion ──
    all_paths = list(args.file) + list(args.files)
    file_contexts = []
    for rp in all_paths:
        try:
            name, content = read_file_for_context(rp)
            file_contexts.append(f"FILE: {name}\n{content}")
        except Exception as e:
            print(f"[{PROG_NAME}] Skipping {rp}: {e}", file=sys.stderr)

    # ── Memory retrieval ──
    memory_text = ""
    if use_memory_trigger:
        hits = search_memories(user_text, k=3)
        if hits:
            memory_text = "\n\n".join(
                [f"### {p.name}\n{cap(txt, 2000)}" for p, txt in hits]
            )

    # ── Build messages ──
    system_prompt = args.system or CONSTITUTION
    messages = [{"role": "system", "content": system_prompt}]

    # Build user message with context
    user_message_parts = []
    if memory_text:
        user_message_parts.append(f"[MEMORY]\n{memory_text}\n[/MEMORY]")
    user_message_parts.append(user_text)
    if file_contexts:
        user_message_parts.extend(file_contexts)

    full_user_message = "\n\n".join(user_message_parts)
    messages.append({"role": "user", "content": full_user_message})

    # ── Log session turn ──
    append_turn(session_file, "USER", user_text)

    # ── Run inference ──
    tool_calls_made = []
    if args.tools:
        answer, tool_calls_made = run_with_tools(
            messages, yolo=args.yolo, debug=args.debug
        )
        print(answer)
    else:
        answer = stream_response(messages)

    # ── Log ──
    append_turn(session_file, "ASSISTANT", cap(answer, 4000))
    log_query(
        prompt=user_text,
        output=answer,
        mode="tools" if args.tools else "standard",
        tool_calls=tool_calls_made,
        files=all_paths or None,
    )

    # ── Memory write ──
    if remember_trigger:
        session_text = session_file.read_text(encoding="utf-8", errors="replace")
        title5, md = make_memory(session_text)
        mem_path = write_memory_file(title5, md)
        print(f"\n[{PROG_NAME}] Wrote memory: {mem_path}")


if __name__ == "__main__":
    main()
```
